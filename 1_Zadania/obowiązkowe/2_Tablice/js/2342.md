function distFromAvarage(numberArray){
var avarage=0;
numberArray.ForEach(element => {
    avarage += element;
});

avarage = avarage / numberArray.length;
console.log(avarage);
var returnArray =[];
numberArray.forEach(element =>{
    returnArray.push(element-avarage);
});

return returnArray;
}
